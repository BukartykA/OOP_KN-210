# Приклад реалізації поліморфізму та інкапсуляції
class Bow(Item):
    def __init__(self, name, attack_power, range_power):
        super().__init__(name)
        self.__attack_power = attack_power # Інкапсуляція
        self.__range_power = range_power

    def attack(self, enemy): # Поліморфізм
        damage = self.__attack_power + randint(5, 15) + self.__range_power
        enemy.health -= damage
        return f"Лук {self.name} стріляє! Нанесено {damage} шкоди."
